**2. JS — add `autoGenerate()` function and debounce timer** (inside `initUI()`, after the existing `updateGenerateButton` function definition, before the `btnGenerate.addEventListener` block):

```js
// --- Auto-generate with debounce ---
var autoGenerateTimer = null;

function autoGenerate() {
  clearTimeout(autoGenerateTimer);
  autoGenerateTimer = setTimeout(function() {
    var ibanVal = fieldIBAN.value.replace(/\s/g, '').toUpperCase();
    var nameVal = fieldName.value.trim();
    // Only auto-generate when minimum required fields are valid
    if (!validateIBAN(ibanVal) || !nameVal) return;

    var amountStr = fieldAmount.value.trim();
    var remittance = fieldRemittance.value.trim();
    var bic = fieldBIC ? fieldBIC.value.trim() : '';
    var amount = amountStr ? parseAmountString(amountStr) : null;

    try {
      var payload = formatEPCPayload({ iban: ibanVal, name: nameVal, amount: amount, bic: bic, remittance: remittance });
      var validation = validateEPCPayload(payload);
      if (!validation.valid) return; // silently skip if payload exceeds limits

      // Flash animation to signal QR updated
      qrOutput.classList.remove('qr-refreshing');
      // Force reflow so removing+re-adding the class triggers the animation
      void qrOutput.offsetWidth;
      qrOutput.classList.add('qr-refreshing');

      renderQR(payload, qrOutput);
      qrSection.classList.remove('hidden');
    } catch (e) {
      // Swallow errors during auto-generate; user can still press the button for an explicit error message
    }
  }, 600); // 600ms debounce — feels instant after a natural pause, not nauseating on keydown
}
```

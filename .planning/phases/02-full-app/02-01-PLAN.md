**Add the following code:**

```javascript
// =============================================================================
// Phase 2: UI wiring — event handlers, validation, QR generation
// =============================================================================

(function initUI() {
  var pasteArea = document.getElementById('paste-area');
  var fieldIBAN = document.getElementById('field-iban');
  var fieldName = document.getElementById('field-name');
  var fieldAmount = document.getElementById('field-amount');
  var fieldRemittance = document.getElementById('field-remittance');
  var fieldBIC = document.getElementById('field-bic');
  var bicToggle = document.getElementById('bic-toggle');
  var bicWrapper = document.getElementById('bic-wrapper');
  var btnGenerate = document.getElementById('btn-generate');
  var qrSection = document.getElementById('qr-section');
  var qrOutput = document.getElementById('qr-output');
  var extractionStatus = document.getElementById('extraction-status');
  var ibanError = document.getElementById('iban-error');
  var ibanStatus = document.getElementById('iban-status');
  var nameStatus = document.getElementById('name-status');
  var amountStatus = document.getElementById('amount-status');
  var remittanceCount = document.getElementById('remittance-count');

  // --- Three-state extraction feedback ---
  function setFieldStatus(statusEl, state, message) {
    statusEl.textContent = message || '';
    statusEl.className = 'field-status';
    if (state) statusEl.classList.add(state);
  }

  function updateExtractionStatus(result) {
    var parts = [];
    var allSuccess = true;
    var anySuccess = false;

    // IBAN status
    if (result.iban) {
      setFieldStatus(ibanStatus, 'success', 'IBAN gevonden');
      anySuccess = true;
    } else {
      setFieldStatus(ibanStatus, 'failed', 'Geen IBAN gevonden');
      allSuccess = false;
    }

    // Name status
    if (result.name) {
      setFieldStatus(nameStatus, 'success', 'Naam gevonden');
      anySuccess = true;
    } else {
      setFieldStatus(nameStatus, 'failed', 'Geen naam gevonden');
      allSuccess = false;
    }

    // Amount status
    if (result.amount !== null) {
      setFieldStatus(amountStatus, 'success', 'Bedrag gevonden');
      anySuccess = true;
    } else {
      setFieldStatus(amountStatus, 'partial', 'Geen bedrag gevonden — vul handmatig in');
      allSuccess = false;
    }

    // Overall extraction status
    if (allSuccess) {
      extractionStatus.textContent = 'Alle velden automatisch ingevuld';
      extractionStatus.className = 'extraction-status';
      extractionStatus.style.color = '#2e7d32';
    } else if (anySuccess) {
      extractionStatus.textContent = 'Gedeeltelijk herkend — controleer de velden';
      extractionStatus.className = 'extraction-status';
      extractionStatus.style.color = '#f9a825';
    } else {
      extractionStatus.textContent = 'Geen betaalgegevens herkend';
      extractionStatus.className = 'extraction-status';
      extractionStatus.style.color = '#d32f2f';
    }
  }

  // --- Paste/input handler: auto-extract ---
  function handlePaste() {
    var text = pasteArea.value.trim();
    if (!text) {
      fieldIBAN.value = '';
      fieldName.value = '';
      fieldAmount.value = '';
      extractionStatus.textContent = '';
      ibanStatus.textContent = '';
      nameStatus.textContent = '';
      amountStatus.textContent = '';
      ibanError.textContent = '';
      updateGenerateButton();
      return;
    }

    var result = parsePaymentText(text);

    // Populate fields
    fieldIBAN.value = result.iban || '';
    fieldName.value = result.name || '';
    fieldAmount.value = result.amount !== null ? result.amount.toFixed(2).replace('.', ',') : '';

    // Show extraction feedback
    updateExtractionStatus(result);

    // Validate IBAN if extracted
    validateIBANField();
    updateGenerateButton();
  }

  pasteArea.addEventListener('input', handlePaste);
  // Also handle paste event for immediate response (input fires after paste completes)
  pasteArea.addEventListener('paste', function() {
    setTimeout(handlePaste, 0);
  });

  // --- IBAN inline validation ---
  function validateIBANField() {
    var val = fieldIBAN.value.replace(/\s/g, '').toUpperCase();
    if (!val) {
      ibanError.textContent = '';
      return false;
    }
    // Basic format check
    if (!/^[A-Z]{2}\d{2}[A-Z0-9]{11,30}$/.test(val)) {
      ibanError.textContent = 'Ongeldig IBAN-formaat';
      return false;
    }
    if (!validateIBAN(val)) {
      ibanError.textContent = 'IBAN controlesom klopt niet (MOD-97)';
      return false;
    }
    ibanError.textContent = '';
    return true;
  }

  fieldIBAN.addEventListener('input', function() {
    validateIBANField();
    updateGenerateButton();
  });

  // --- Name field listener ---
  fieldName.addEventListener('input', function() {
    updateGenerateButton();
  });

  // --- Remittance character counter ---
  fieldRemittance.addEventListener('input', function() {
    remittanceCount.textContent = fieldRemittance.value.length + '/140';
  });

  // --- BIC toggle ---
  bicToggle.addEventListener('click', function() {
    var isHidden = bicWrapper.classList.contains('hidden');
    if (isHidden) {
      bicWrapper.classList.remove('hidden');
      bicToggle.textContent = '- BIC/SWIFT verbergen';
      fieldBIC.focus();
    } else {
      bicWrapper.classList.add('hidden');
      bicToggle.textContent = '+ BIC/SWIFT toevoegen';
    }
  });

  // --- Generate button state ---
  function updateGenerateButton() {
    var ibanVal = fieldIBAN.value.replace(/\s/g, '').toUpperCase();
    var nameVal = fieldName.value.trim();
    var ibanValid = ibanVal.length > 0 && validateIBAN(ibanVal);
    var namePresent = nameVal.length > 0;
    btnGenerate.disabled = !(ibanValid && namePresent);
  }

  // --- Generate QR ---
  btnGenerate.addEventListener('click', function() {
    var ibanVal = fieldIBAN.value.replace(/\s/g, '').toUpperCase();
    var nameVal = fieldName.value.trim();
    var amountStr = fieldAmount.value.trim();
    var remittance = fieldRemittance.value.trim();
    var bic = fieldBIC ? fieldBIC.value.trim() : '';

    // Parse amount: handle both comma and dot as decimal separator
    var amount = null;
    if (amountStr) {
      amount = parseAmountString(amountStr);
    }

    try {
      var payload = formatEPCPayload({
        iban: ibanVal,
        name: nameVal,
        amount: amount,
        bic: bic,
        remittance: remittance
      });

      var validation = validateEPCPayload(payload);
      if (!validation.valid) {
        alert('EPC payload ongeldig: ' + validation.errors.join(', '));
        return;
      }

      renderQR(payload, qrOutput);
      qrSection.classList.remove('hidden');
    } catch (e) {
      alert('Fout bij genereren: ' + e.message);
    }
  });
})();
```

**Part D: Add Copy-to-Clipboard JavaScript**

Add the following code inside the `initUI` IIFE, after the Generate button click handler (before the closing `})();`):

```javascript
  // --- Copy QR to clipboard ---
  var btnCopy = document.getElementById('btn-copy');
  var copyFeedback = document.getElementById('copy-feedback');

  btnCopy.addEventListener('click', function() {
    var svgEl = qrOutput.querySelector('svg');
    if (!svgEl) return;

    // Convert SVG to canvas to blob for clipboard
    var svgData = new XMLSerializer().serializeToString(svgEl);
    var svgBlob = new Blob([svgData], { type: 'image/svg+xml;charset=utf-8' });
    var url = URL.createObjectURL(svgBlob);
    var img = new Image();
    img.onload = function() {
      var canvas = document.createElement('canvas');
      // Render at 400x400 for good quality
      canvas.width = 400;
      canvas.height = 400;
      var ctx = canvas.getContext('2d');
      ctx.fillStyle = '#ffffff';
      ctx.fillRect(0, 0, 400, 400);
      ctx.drawImage(img, 0, 0, 400, 400);
      URL.revokeObjectURL(url);

      canvas.toBlob(function(blob) {
        if (!blob) {
          copyFeedback.textContent = 'Kopiëren mislukt';
          copyFeedback.classList.add('visible');
          setTimeout(function() { copyFeedback.classList.remove('visible'); }, 2000);
          return;
        }
        navigator.clipboard.write([
          new ClipboardItem({ 'image/png': blob })
        ]).then(function() {
          copyFeedback.textContent = 'Gekopieerd!';
          copyFeedback.classList.add('visible');
          setTimeout(function() { copyFeedback.classList.remove('visible'); }, 2000);
        }).catch(function() {
          copyFeedback.textContent = 'Kopiëren mislukt — probeer rechtermuisklik';
          copyFeedback.classList.add('visible');
          setTimeout(function() { copyFeedback.classList.remove('visible'); }, 3000);
        });
      }, 'image/png');
    };
    img.src = url;
  });
```

Implementation notes:
- The SVG-to-canvas pipeline is required because `navigator.clipboard.write` needs a PNG blob, not SVG
- Canvas renders at 400x400 for quality when pasted elsewhere
- White background is explicitly drawn on canvas before SVG (QR codes need white background)
- Clipboard API requires secure context (HTTPS or localhost) — file:// may not support it, the catch handler shows a fallback message
- Feedback text fades in/out using CSS opacity transition
